使用sandman和gunicorn构建高性能REST API
Thu 30 January 2014

sandman 是一个基于 sqlalchemy 和 flask 的简单地将 mysql table映射成 REST API 接口, 使用方法非常简单

如下:

from sandman import app, db
from werkzeug.contrib.fixers import ProxyFix

app.config['SQLALCHEMY_DATABASE_URI'] = 'mysql+mysqldb://timger:timger@127.0.0.1:3306/apiserver'

from sandman.model import register, Model
from sandman.model import activate

class Tweet(Model):
    __tablename__ = 'tweetapi_tweet'

class Apps(Model):
    __tablename__ = 'recommend_apps'

class Products(Model):
    __tablename__ = 'recommend_products'

register((Tweet,))

activate()

app.wsgi_app = ProxyFix(app.wsgi_app)
app.run(debug=True)
